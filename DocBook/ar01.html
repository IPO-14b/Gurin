Пример 1: Следование квадарата за мышкой

Инициализируем переменную окружения.

Получаем доступ к контролю мышки.

Получаем координаты мышки в данный момент.

Отрисовываем квадратик.

Устанавливаем парметры для квадратика.

 
var pjs = new PointJS('2D', 800, 600, { backgroundColor: '#DDDDDD' });
pjs.mouseControl.initMouseControl();
var mp = pjs.mouseControl.getPosition();
pjs.brush.drawRoundRect({
x : mp.x - 12.5,
y : mp.y - 12.5,    
w : 25, h : 25,
radius : 10,
fillColor : '#515151',
});
